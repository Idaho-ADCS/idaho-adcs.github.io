10void state_machine_transition(uint8_t mode)
11{
12 uint8_t curr_mode = CMD_STANDBY; // standby by default
13 // get the current state to compare against
14 xQueuePeek(modeQ, &curr_mode, 0);
15
16 #if DEBUG
17 char debug_str[16];
18 #endif
19
20 // make sure we are entering a new state
21 if (mode == curr_mode) // if not, exit
22 {
23 return;
24 }
25 // stop driving the flywhl any time system mode changes
26 flywhl.stop();
27 bool command_is_valid = true;
28 // change actuator state, set global state variables if needed
29 switch (mode)
30 {
31 case CMD_HEARTBEAT: // transmit sensor readings and state to TES regularly
32 #if DEBUG
33 SERCOM_USB.print("[mode switch]\tEntering HEARTBEAT mode\r\n");

35 break;
36
37 // WARNING: if you switch out of a test mode the test will still be on the stack

39 case CMD_TST_BASIC_MOTION: // loop flywhl PWM signal 0-100%
40 case CMD_TST_BASIC_AD:
41 case CMD_TST_BASIC_AC:
42 case CMD_TST_SIMPLE_DETUMBLE: // use PID to stop rotation about Z axis
43 case CMD_TST_SIMPLE_ORIENT: // use photodiode input to point X+ side at light

45 #if DEBUG

47 #endif
48 break;

51 #if DEBUG

57 case CMD_ORIENT_X_POS:
58 case CMD_ORIENT_Y_POS:
59 case CMD_ORIENT_X_NEG:

61 #if DEBUG

73 SERCOM_USB.print("\r\n");
74 #endif

128// @brief get direction the adcs should turn to align X+ with the light source
129//
130// @param[in] vals The sensor values, unscaled
131//
132// @return The direction to turn the adcs (clockwise or counter-clockwise)
133//
134MotorDirection getDirection(PDdata vals){
135 uint8_t max=0; // channel receiving the most light
136 float max_val=-1;
137 // calculate side/channel receiving most light, ignore +/-Z so only look at first 4 values
138 for(int i=0; i < 4; i++){
139 if(vals.data[i] >= max_val){
140 // save channel and value
141 max_val = vals.data[i];
142 max = i;
143 }
144 }
145
146 #if DEBUG
147 SERCOM_USB.print("getDirection found max on channel = ");
148 SERCOM_USB.println(max);
149 #endif
150
151 // translate to coordinates and calculate direction to spin
152 switch(max){
153 case X_POS:
154 return IDLE;
155 case X_NEG:
156 return CW;
157 case Y_POS:
158 return CCW;
159 case Y_NEG:
160 return CW;
161 case Z_POS:
162 case Z_NEG:
163 return IDLE;
164 }

508void simple_detumble(void *pvParameters)
509{
510#if DEBUG
511 SERCOM_USB.print("[basic detumbl]\tTask started\r\n");
512#endif
513
514 uint8_t mode; // last received ADCS mode
515
516 const int target_rot_vel = 0; // rotational velocity we want to maintain
517 int num_steps = 1;
518
519 const float P = 1; // proportional component
520 int I = 0; // integral component
521 int D = 0; // derivative component
522
523 int prev_error = 0; // error from last loop iteration
524 int error = 0; // assumed stationary at start
525
526 int pwm_output = 0; // init at zero, signed to represent direction
527
528 IMUdata imu;
529
530 while (true)
531 {
532 // #if DEBUG
533 // SERCOM_USB.print("[basic detumbl]\tChecked mode\r\n");
534 // #endif
535 xQueuePeek(modeQ, &mode, 0);
536
537 if (mode == CMD_TST_SIMPLE_DETUMBLE)
538 {
539 xSemaphoreTake(IMUsemphr, portMAX_DELAY);
540 // calculate error
541 xQueuePeek(IMUq, &imu, 0);
542 float rot_vel_z = imu.gyrZ;
543
544 error = rot_vel_z - target_rot_vel; // difference between current state and target state
545 // proportional term calculation
546 float p_term = error * P;
547 // integral term accumulation
548 I += error;
549 int i_term = I;
550 // derivative term
551 int d_term = I / num_steps;
552 // output
553 pwm_output = p_term; //+ i_term + d_term; /* add other terms when initial tuning complete */
554 if (pwm_output > 255)
555 { // cap output
556 pwm_output = 255;
557 }
558
559 prev_error = error;
560 num_steps++;
561
562 // unsigned pwm to motor with direction
563 if (error > 0)
564 {
565 flywhl.run(CW, abs(pwm_output));
566 }
567 else if (error < 0)
568 {
569 flywhl.run(CCW, abs(pwm_output));
570 }
571 else
572 {
573 // TODO: desaturate with magnetorquers if |pwm_output| == 255 or in state of equilibrium
574 // for the meantime, just keep doing the same thing
575 }
576
577 #if DEBUG
578 SERCOM_USB.print("[basic detumbl]\t====== PID LOOP ======\r\n");
579 SERCOM_USB.print("\t\tIMU VELOCITY = ");
580 SERCOM_USB.print(rot_vel_z);
581 SERCOM_USB.print(" degrees/sec\r\n");
582
583 SERCOM_USB.print("\t\tERROR = ");
584 SERCOM_USB.print(error);
585 SERCOM_USB.print("\r\n");
586
587 SERCOM_USB.print("\t\tPWM OUTPUT = ");
588 SERCOM_USB.print(abs(pwm_output));
589 SERCOM_USB.print("\r\n\t\t======================\r\n");
590 #endif
591
592 prev_error = error;
593 }
594 vTaskDelay(pdMS_TO_TICKS(10));
595 }
