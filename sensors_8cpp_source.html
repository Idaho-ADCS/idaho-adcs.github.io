194void readIMU(void *pvParameters)
195{
196 ICM_20948_I2C *sensor_ptr1 = &IMU1;
197 #if NUM_IMUS >= 2
198 ICM_20948_I2C *sensor_ptr2 = &IMU2;

203 const int DECIMATION = 4;
204 const int NUM_DECIMATIONS = 8;
205
206 float gyrXavgs[NUM_DECIMATIONS];
207 float gyrYavgs[NUM_DECIMATIONS];
208 float gyrZavgs[NUM_DECIMATIONS];
209
210 float gyrXresult = 0.0f;
211 float gyrYresult = 0.0f;
212 float gyrZresult = 0.0f;
213
214 int readcntr = 0;
215 int avgcntr = 0;
216
217 result.magX = 0.0f;
218 result.magY = 0.0f;
219 result.magZ = 0.0f;
220 result.gyrX = 0.0f;
221 result.gyrY = 0.0f;
222 result.gyrZ = 0.0f;
223
224 for (int i = 0; i < NUM_DECIMATIONS; i++)
225 {
226 gyrXavgs[i] = 0.0f;
227 gyrYavgs[i] = 0.0f;
228 gyrZavgs[i] = 0.0f;
229 }
230
231 while (1)
232 {
233 #if NUM_IMUS >= 2
234 if (IMU1.dataReady() && IMU2.dataReady())
235 {
236 #else
237 if (IMU1.dataReady())
238 {
239 #endif
240 xSemaphoreTake(IMUsemphr, 0);
241 IMU1.getAGMT();
242 #if NUM_IMUS >= 2
243 IMU2.getAGMT();
244 #endif
245 xSemaphoreGive(IMUsemphr);
246
247 result.magX = sensor_ptr1->magX();
248 result.magY = sensor_ptr1->magY();
249 result.magZ = sensor_ptr1->magZ();
250
251 #if NUM_IMUS >= 2
252 gyrXavgs[avgcntr] += ((sensor_ptr1->gyrX() + sensor_ptr2->gyrX()) / 2);
253 gyrYavgs[avgcntr] += ((sensor_ptr1->gyrY() + sensor_ptr2->gyrY()) / 2);
254 gyrZavgs[avgcntr] += ((sensor_ptr1->gyrZ() + sensor_ptr2->gyrZ()) / 2);
255 #else
256 gyrXavgs[avgcntr] += sensor_ptr1->gyrX();
257 gyrYavgs[avgcntr] += sensor_ptr1->gyrY();
258 gyrZavgs[avgcntr] += sensor_ptr1->gyrZ();
259 #endif
260
261 readcntr++;
262
263 if (readcntr >= DECIMATION)
264 {
265 gyrXavgs[avgcntr] /= (float)DECIMATION;
266 gyrYavgs[avgcntr] /= (float)DECIMATION;
267 gyrZavgs[avgcntr] /= (float)DECIMATION;
268
269 for (int i = 0; i < NUM_DECIMATIONS; i++)
270 {
271 gyrXresult += gyrXavgs[i];
272 gyrYresult += gyrYavgs[i];
273 gyrZresult += gyrZavgs[i];
274 }
275
276 gyrXresult /= (float)NUM_DECIMATIONS;
277 gyrYresult /= (float)NUM_DECIMATIONS;
278 gyrZresult /= (float)NUM_DECIMATIONS;
279
280 result.gyrX = gyrXresult;
281 result.gyrY = gyrYresult;
282 result.gyrZ = gyrZresult;
283
284 avgcntr = (avgcntr + 1) % NUM_DECIMATIONS;
285
286 gyrXavgs[avgcntr] = 0.0f;
287 gyrYavgs[avgcntr] = 0.0f;
288 gyrZavgs[avgcntr] = 0.0f;
289
290 readcntr = 0;
291 }
292 }
293
294 xQueueOverwrite(IMUq, &result);
295
296 vTaskDelay(5 / portTICK_PERIOD_MS);
297 }

412void printFormattedFloat(float val, uint8_t leading, uint8_t decimals)
413{
414 float aval = abs(val);
415 if (val < 0)
416 {
417 SERCOM_USB.print("-");
418 }
419 else
420 {
421 SERCOM_USB.print(" ");
422 }
423 for (uint8_t indi = 0; indi < leading; indi++)
424 {
425 uint32_t tenpow = 0;
426 if (indi < (leading - 1))
427 {
428 tenpow = 1;
429 }
430 for (uint8_t c = 0; c < (leading - 1 - indi); c++)
431 {
432 tenpow *= 10;
433 }
434 if (aval < tenpow)
435 {
436 SERCOM_USB.print("0");
437 }
438 else
439 {
440 break;
441 }
442 }
443 if (val < 0)
444 {
445 SERCOM_USB.print(-val, decimals);
446 }
447 else
448 {
449 SERCOM_USB.print(val, decimals);
450 }
451}
